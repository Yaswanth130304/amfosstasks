a, b = map(int, input().split())
c = b // a
if a == b:
    print(0)
elif c < 2 or b % a != 0:
    print(-1)
else:
    t = 0
    while c % 2 == 0:
        t += 1
        c //= 2
    while c % 3 == 0:
        t += 1
        c //= 3
    if c > 1:
        print(-1)
    else:
        print(t)
